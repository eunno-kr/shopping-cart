# 장바구니 상품 담기

장바구니 = {}
매출기록 = []

장바구니물품갯수 = random.randint(1, 15)

def 장바구니_담기(상품명, 수량):
    if 상품명 not in 물품목록:
        print(f"{상품명}은 재고가 없는 상품입니다.")
        return

    if 물품목록[상품명]["재고"] < 수량:
        print(f"재고부족! (현재{상품명}재고: {물품목록[상품명]['재고']}개)")
        return
    장바구니[상품명] = 장바구니.get(상품명, 0) + 수량
    print(f"{상품명}이(가) {수량}개 장바구니에 담겼습니다.")

#장바구니 계산
def 장바구니_계산():
    if not 장바구니:
        print("장바구니가 비어 있습니다.")
        return 0
    합계=0
    총수량=0
    print("장바구니 목록")
    print("  " + "-" * 38)
    for 상품명, 수량 in 장바구니.items():
        단가 = 물품목록[상품명]["단가"]
        소계 = 단가 * 수량
        합계 += 소계
        총수량 += 수량
        print (f"{상품명}{수량}개 * {단가:,}원 = {소계:,}원")
    print("-" * 40)
    print(f"총 수량: {총수량}개" )
    print(f"총 금액: {합계:,}원" )
    return 합계

#할인 적용 계산
def 할인적용(할인율):
    합계 = 장바구니_계산()

    최종금액 = 합계 - 할인금액

    print(f"최종 금액: {최종금액:,}원")
    return 최종금액

def 결재(결재방법="현금" ):

    오늘 = datetime.now(). strftime("%Y-%m-%d")
    for 상품명, 수량  in 장바구니.items():

            "상품명": 상품명,
            "수량": 수량,
            "금액": 물품목록[상품명]["단가"]*수량
        })
    장바구니.clear()
    print("결재 완료! 영수증을 출력합니다.")
    영수증출력(합계, 결재방법)

#재고 몇개 이하 재고 알림
def 재고알림(기준=10):
    재고부족알림 = [
        (상품명, 정보["재고"])
        for 상품명, 정보 in 물품목록.items()
        if 정보["재고"]<=기준
    ]
    if not 재고부족알림:
        print(f"재고{기준}개 이하 상품 없음")
        return
    print(f"재고 {기준}개 이하 상품")
    print("-" * 25)
    for 상품명, 재고 in 재고부족알림:
        print(f"{상품명:12}{재고}개 남음")

#일간매출
#주간매출
#월간매출
def 매출집계(기간="일"):
    오늘 = datetime.now()
    합계 = 0
    for 기록 in 매출기록:
        날짜 = datetime.strptime(기록["날짜"],"%Y-%m-%d")
        포함 = False
        if 기간 == "일" and 날짜.date() == 오늘.date():
            포함 = True
        elif 기간 == "주" and (오늘 - 날짜).days <= 7:
            포함 = True
        elif 기간 == "월" and 날짜.month == 오늘.month:
            포함 = True
        if 포함:
            합계 += 기록["금액"]
        print(f"{기간}간 매출: {합계:,}원")
        return 합계

#베스트 및 워스트 할인 프로모션
def 베스트워스트():
    판매량 = {}
    for 기록 in 매출기록:
        판매량[기록["상품명"]] = 판매량.get(기록["상품명"],0) + 기록["수량"]
    if not 판매량:
        print("아직 판매 기록이 없습니다.")
        return
    정렬 = sorted(판매량.items(), key=lambda x:x[1], reverse=True)
    print(f"베스트: {정렬[0][0]}({정렬[0][1]}개)")
    print(f"워스트: {정렬[-1][0]}({정렬[-1][1]}개)")
